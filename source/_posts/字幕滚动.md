### js

```js
$(function() {
  setInterval(function() {
    $('p')
      .first()
      .css({ marginTop: '-24px' });
    setTimeout(function() {
      $('.box').append($('p').first());
      $('p')
        .last()
        .removeAttr('style');
    }, 1000);
  }, 3000);
});
```
